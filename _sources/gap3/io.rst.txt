File I/O
========

.. code-block:: none

	Print("Hello World!");
	LogTo("log.txt");			# log to log.txt
	PrintLine("Numbers: ", 1, ", ", 2);
	PrintTo("file.txt", "a := ", 2);
	AppendTo("file.txt", ";\n");
	Read("file.txt");			# read file into stdin
	Exec("del file.txt");			# MS/DOS system call
	a;					# file contents was executed
	LogTo();				# Turn off logging
